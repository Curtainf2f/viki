# 常指针

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    int a = 1, b = 2;
    const int *p1 = &a; // 访问常量的指针
    // (*p1) = 2; // error: assignment of read-only location '* p'
    p1 = &b; // ok

    int const *p2 = &a; // 访问常量的指针
    // (*p2) = 2; // error: assignment of read-only location '* p2'
    p2 = &b; // ok

    int * const p3 = &a; // 访问变量的常指针
    (*p3) = 2; // ok
    // p3 = &b; // error: assignment of read-only variable 'p3'

    const int * const p4 = &a; // 访问常量的常指针
    // (*p4) = 2; // error: assignment of read-only location '*(const int*)p4'
    // p4 = &b; // error: assignment of read-only variable 'p4'

    return 0;
}
```
