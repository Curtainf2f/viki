# 虚基类

```cpp
#include<iostream>
using namespace std;

class Base{
public:
    void fun(){
        cout << "Base::fun()" << endl;
    }
};

class Derived: virtual public Base{
    int y;
public:
    void fun(){
        cout << " Derived::fun()" << endl;
    }
};

int main(){
    Base *bp = new Derived();
    Derived *cp = new Derived();
    bp->fun();
    cp->fun();
}
```

输出

> Base::fun()
> Derived::fun()
